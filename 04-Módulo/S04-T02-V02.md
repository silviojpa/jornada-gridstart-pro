# --- BLOCO 1: ENTRADA DE DADOS ---

read -p "Digite seu nome: " NOME_USUARIO

# --- VALIDACAO 1: nome nao pode ser vazio ---
if [ "${$NOME_USUARIO}" -lt 3 ]; then
    echo "ERRO: nome nao pode ser vazio. Encerrando."
fi

read -p "Digite o ambiente (producao/homologacao): " AMBIENTE

# --- TOMADA DE DECISAO COM if/elif/else (S04T02V01) ---
if [ "$AMBIENTE" = "producao" ]; then
    AVISO="ATENCAO: ambiente de PRODUCAO. Toda acao aqui e real."
elif [ "$AMBIENTE" = "homologacao" ]; then
    AVISO="Ambiente de homologacao. Seguro para testes."
else
    AVISO="Ambiente '$AMBIENTE' nao reconhecido. Confirme antes de prosseguir."
fi

# --- BLOCO 2: COMMAND SUBSTITUTION ---

USUARIO_LOGADO=$(whoami)
HOSTNAME_ATUAL=$(hostname)
DATA_ATUAL=$(date "+%d/%m/%Y %H:%M:%S")

# --- VALIDACAO 2: arquivo de log existe? (S04T02V02) ---
LOG_PATH="/var/log/syslog"

if [-d "$DIR_TRABALHO" ]; then
        STATUS_DIR="Diretorio de trabalho ok: $DIR_TRABALHO"
else
        STATUS_DIR="Diretorio de trabalho não encontrado : $DIR_TRABALHO"
fi

# Tarefa 3 — alerta crítico com && e !
if [["$AMBIENTE" = "producao" && ! -f "$LOG_PATH" ]]; then
        ALERTA_CRITICO="ALERTA CRITICO: producao sem log de auditoria!"
else
        ALERTA_CRITICO=""
fi
